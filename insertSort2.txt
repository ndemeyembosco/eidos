length = 1000;
range = 1000;
x = 1.0:length;
print("Initial x:");
print(x);
for ( iter in 1:length ) {
  index = randInt(length);
  value = randFloat(range); 
  x[index] = value;
}
print("Randomized x:");
print(x);
bound = length-1;
for ( acc in 1:bound ){
  key = x[acc];
  j = acc - 1;
  while (j >= 0 & x[j] > key){
    x[j+1] = x[j];
    j = j-1;
  }
  x[j+1] = key;
}
print("Ordered x");
print(x);
